JavaScript 排序算法揭秘

JavaScript 排序算法揭秘

🌟【程序员必备】揭秘 JavaScript 排序算法那些事儿！💻✨

哈喽各位小伙伴～今天给大家分享四种常见的排序算法： TimSort、QuickSort、HeapSort 和 MergeSort。每种算法都有独特的小技巧，快来get代码和原理， 面试排序问题再也不慌张啦！🎉🚀

1. TimSort —— 归并 + 插入排序的完美组合 💡🔀

TimSort 是现代引擎（比如 Python 的 list.sort() 和 JS V8 部分场景）常用的排序算法。 它先用插入排序处理小区间，再用归并排序合并区间。超适合处理部分有序数据哦～💪 

function insertionSort(arr, left, right) {
  for (let i = left + 1; i <= right; i++) {
    let temp = arr[i];
    let j = i - 1;
    while (j >= left && arr[j] > temp) {
      arr[j + 1] = arr[j];
      j--;
    }
    arr[j + 1] = temp;
  }
}

function merge(arr, left, mid, right) {
  let leftArr = arr.slice(left, mid + 1);
  let rightArr = arr.slice(mid + 1, right + 1);
  let i = 0, j = 0, k = left;
  while (i < leftArr.length && j < rightArr.length) {
    arr[k++] = (leftArr[i] <= rightArr[j]) ? leftArr[i++] : rightArr[j++];
  }
  while (i < leftArr.length) arr[k++] = leftArr[i++];
  while (j < rightArr.length) arr[k++] = rightArr[j++];
}

function timSort(arr) {
  const RUN = 32;
  const n = arr.length;
  // 先对小区间使用插入排序
  for (let i = 0; i < n; i += RUN) {
    insertionSort(arr, i, Math.min(i + RUN - 1, n - 1));
  }
  // 再两两归并
  for (let size = RUN; size < n; size *= 2) {
    for (let left = 0; left < n; left += 2 * size) {
      let mid = left + size - 1;
      let right = Math.min(left + 2 * size - 1, n - 1);
      if (mid < right) merge(arr, left, mid, right);
    }
  }
}

let arr1 = [5, 21, 7, 23, 19];
timSort(arr1);
console.log("TimSort:", arr1);  // 输出排序后的数组 🎯

2. QuickSort —— 分治法的经典代表 ⚡️✂️

QuickSort 利用分治思想，选取一个 pivot， 将数组分为左右两部分，再递归排序。这是一种直观且高效的方法哦～🔥 

function quickSort(arr) {
  if (arr.length <= 1) return arr;
  let pivot = arr[Math.floor(arr.length / 2)];
  let left = arr.filter(x => x < pivot);
  let middle = arr.filter(x => x === pivot);
  let right = arr.filter(x => x > pivot);
  return [...quickSort(left), ...middle, ...quickSort(right)];
}

let arr2 = [3, 6, 8, 10, 1, 2, 1];
console.log("QuickSort:", quickSort(arr2));  // 快速排序结果 🏎️

3. HeapSort —— 利用堆结构进行排序 ⛰️📦

HeapSort 通过构建最大堆，每次将最大的元素交换到数组末尾，逐步完成排序。 时间复杂度稳定，适合需要保证 O(n log n) 的场景～👌 

function heapify(arr, n, i) {
  let largest = i;
  let left = 2 * i + 1;
  let right = 2 * i + 2;
  if (left < n && arr[left] > arr[largest]) largest = left;
  if (right < n && arr[right] > arr[largest]) largest = right;
  if (largest !== i) {
    [arr[i], arr[largest]] = [arr[largest], arr[i]];
    heapify(arr, n, largest);
  }
}

function heapSort(arr) {
  const n = arr.length;
  // 构建最大堆
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i);
  }
  // 不断提取堆顶最大元素
  for (let i = n - 1; i > 0; i--) {
    [arr[0], arr[i]] = [arr[i], arr[0]];
    heapify(arr, i, 0);
  }
}

let arr3 = [12, 11, 13, 5, 6, 7];
heapSort(arr3);
console.log("HeapSort:", arr3);  // 排序结果展示 🎊

4. MergeSort —— 归并排序，稳定性满分 💖🔗

MergeSort 采用分治思想，将数组不断拆分成单个元素，再归并成有序序列。 它是稳定排序的经典算法，适合对稳定性要求高的场景哦～🌈 

function mergeSort(arr) {
  if (arr.length <= 1) return arr;
  let mid = Math.floor(arr.length / 2);
  let left = mergeSort(arr.slice(0, mid));
  let right = mergeSort(arr.slice(mid));
  return mergeArrays(left, right);
}

function mergeArrays(left, right) {
  let result = [];
  let i = 0, j = 0;
  while (i < left.length && j < right.length) {
    result.push(left[i] < right[j] ? left[i++] : right[j++]);
  }
  return result.concat(left.slice(i)).concat(right.slice(j));
}

let arr4 = [38, 27, 43, 3, 9, 82, 10];
console.log("MergeSort:", mergeSort(arr4));  // 最终排序结果 📌

💡 小结

  • TimSort：适合部分有序数据，结合插入排序与归并排序（稳定排序）✅
  • QuickSort：分治法代表，简单高效，但最坏情况需注意⚠️
  • HeapSort：利用堆结构，每次提取最大元素，时间复杂度稳定🔒
  • MergeSort：经典归并，稳定性满分，适用于要求高稳定性的场景🎯
